165 bool Camera::runCalibration( vector<vector<Point2f> > imagePoints,
166  Size imageSize, Size boardSize,
167  float squareSize, float aspectRatio,
168  int flags, Mat& cameraMatrix, Mat& distCoeffs,
169  vector<Mat>& rvecs, vector<Mat>& tvecs,
170  vector<float>& reprojErrs,
171  double& totalAvgErr)
172 {
173  cameraMatrix = Mat::eye(3, 3, CV_64F);
174  if( flags & CV_CALIB_FIX_ASPECT_RATIO )
175  cameraMatrix.at<double>(0,0) = aspectRatio;
176 
177  distCoeffs = Mat::zeros(4, 1, CV_64F);
178 
179  vector<vector<Point3f> > objectPoints(1);
180  calcChessboardCorners(boardSize, squareSize, objectPoints[0]);
181 
182  objectPoints.resize(imagePoints.size(),objectPoints[0]);
183 
184  double rms = calibrateCamera(objectPoints, imagePoints, imageSize, cameraMatrix,
185  distCoeffs, rvecs, tvecs,CV_CALIB_FIX_K3);
186  printf("RMS error reported by calibrateCamera: %g\n", rms);
187 
188  bool ok = checkRange(cameraMatrix) && checkRange(distCoeffs);
189 
190  totalAvgErr = computeReprojectionErrors(objectPoints, imagePoints,
191  rvecs, tvecs, cameraMatrix, distCoeffs, reprojErrs);
192 
193  return ok;
194 }

206 double Camera::computeReprojectionErrors(
207  const vector<vector<Point3f> >& objectPoints,
208  const vector<vector<Point2f> >& imagePoints,
209  const vector<Mat>& rvecs, const vector<Mat>& tvecs,
210  const Mat& cameraMatrix, const Mat& distCoeffs,
211  vector<float>& perViewErrors )
212 {
213  vector<Point2f> imagePoints2;
214  int i, totalPoints = 0;
215  double totalErr = 0, err;
216  perViewErrors.resize(objectPoints.size());
217 
218  for( i = 0; i < (int)objectPoints.size(); i++ )
219  {
220  projectPoints(Mat(objectPoints[i]), rvecs[i], tvecs[i],
221  cameraMatrix, distCoeffs, imagePoints2);
222  err = norm(Mat(imagePoints[i]), Mat(imagePoints2), CV_L2);
223  int n = (int)objectPoints[i].size();
224  perViewErrors[i] = (float)std::sqrt(err*err/n);
225  totalErr += err*err;
226  totalPoints += n;
227  }
228 
229  return std::sqrt(totalErr/totalPoints);
230 }
